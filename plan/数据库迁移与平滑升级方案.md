# 数据库迁移与平滑升级方案

## 一、核心设计原则

### 1.1 原则
- **向后兼容**：新版本必须能读取旧版本数据库
- **无感知升级**：用户无需手动操作，升级自动完成
- **可回滚**：升级失败能恢复到升级前状态
- **增量迁移**：每个版本只负责自己的变更，不越级处理

### 1.2 版本号策略
使用语义化版本号，数据库版本与应用版本解耦：
```
应用版本: 1.0.0 → 数据库版本: 1
应用版本: 1.1.0 → 数据库版本: 2 (新增标签功能)
应用版本: 1.2.0 → 数据库版本: 3 (新增目标系统)
```

数据库版本存储在 `user_config` 表的 `db_schema_version` 键中。

## 二、数据库迁移实现方案

### 2.1 迁移流程架构

```
应用启动
    ↓
初始化 SQLite 连接
    ↓
读取当前 db_schema_version（无则为 v1）
    ↓
目标版本 = 当前代码支持的最新版本
    ↓
如果 当前版本 < 目标版本
    ↓
    ┌─ 备份数据库到 study_time.db.backup_v{N}
    ├─ 循环执行迁移脚本: v1→v2, v2→v3, ...
    ├─ 每个迁移成功后更新 db_schema_version
    └─ 全部成功 → 继续启动
    ↓
如果 迁移失败
    ↓
    ├─ 从备份恢复数据库
    ├─ 弹窗提示用户："升级失败，已恢复原有数据"
    └─ 应用正常启动（旧功能可用）
```

### 2.2 迁移脚本目录结构

```
plan/
└── database-migration.md    # 本文档
sql/
├── migrations/
│   ├── v001_initial.js      # 初始表创建（v1）
│   ├── v002_add_tags.js     # 标签系统（v1→v2）
│   ├── v003_add_goals.js    # 目标系统（v2→v3）
│   └── v004_add_badges.js   # 徽章系统（v3→v4）
├── create_study_sessions_table.sql   # 原有文件
└── create_user_config_table.sql      # 原有文件
```

### 2.3 迁移脚本模板

每个迁移脚本必须包含 3 个函数：
```javascript
// v002_add_tags.js
module.exports = {
  // 目标版本号
  version: 2,
  
  // 描述
  description: "新增学习标签分类系统",
  
  // 升级执行
  up: async (db) => {
    await runSql(db, `
      CREATE TABLE study_tags (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL UNIQUE,
        color TEXT DEFAULT '#007AFF',
        created_at TEXT DEFAULT CURRENT_TIMESTAMP
      )
    `);
    
    await runSql(db, `
      CREATE TABLE session_tags (
        session_id INTEGER,
        tag_id INTEGER,
        PRIMARY KEY (session_id, tag_id),
        FOREIGN KEY (session_id) REFERENCES study_sessions(id),
        FOREIGN KEY (tag_id) REFERENCES study_tags(id)
      )
    `);
    
    await runSql(db, `
      CREATE INDEX IF NOT EXISTS idx_study_sessions_start_time 
      ON study_sessions(start_time)
    `);
  },
  
  // 降级回滚（可选，但强烈建议实现）
  down: async (db) => {
    await runSql(db, "DROP TABLE IF EXISTS session_tags");
    await runSql(db, "DROP TABLE IF EXISTS study_tags");
  }
};

// 通用 SQL 执行 Promise 包装
function runSql(db, sql, params = []) {
  return new Promise((resolve, reject) => {
    db.run(sql, params, function(err) {
      if (err) reject(err);
      else resolve(this);
    });
  });
}
```

## 三、主进程集成代码

### 3.1 迁移管理器 `database-migrator.js`

```javascript
const path = require('path');
const fs = require('fs');

// 按顺序排列的所有迁移脚本
const migrations = [
  require('./sql/migrations/v001_initial.js'),
  require('./sql/migrations/v002_add_tags.js'),
  require('./sql/migrations/v003_add_goals.js'),
  require('./sql/migrations/v004_add_badges.js'),
];

// 获取当前数据库版本
async function getCurrentVersion(db) {
  return new Promise((resolve) => {
    db.get(
      "SELECT value FROM user_config WHERE key = 'db_schema_version'",
      (err, row) => {
        if (err || !row) {
          // user_config 表可能还不存在（v1）
          resolve(1);
        } else {
          resolve(parseInt(row.value) || 1);
        }
      }
    );
  });
}

// 更新数据库版本
async function setVersion(db, version) {
  return new Promise((resolve, reject) => {
    db.run(
      "INSERT OR REPLACE INTO user_config(key, value) VALUES (?, ?)",
      ['db_schema_version', version.toString()],
      (err) => err ? reject(err) : resolve()
    );
  });
}

// 备份数据库
async function backupDatabase(dbPath, version) {
  const backupPath = `${dbPath}.backup_v${version}`;
  fs.copyFileSync(dbPath, backupPath);
  return backupPath;
}

// 从备份恢复
async function restoreFromBackup(dbPath, backupPath) {
  if (fs.existsSync(backupPath)) {
    fs.copyFileSync(backupPath, dbPath);
    return true;
  }
  return false;
}

// 主迁移函数
async function runMigrations(db, dbPath) {
  const currentVersion = await getCurrentVersion(db);
  const targetVersion = migrations.length;
  
  console.log(`[DB Migration] 当前版本: v${currentVersion}, 目标版本: v${targetVersion}`);
  
  if (currentVersion >= targetVersion) {
    console.log("[DB Migration] 数据库已是最新版本，无需迁移");
    return { success: true, isLatest: true };
  }
  
  // 迁移前备份
  const backupPath = await backupDatabase(dbPath, currentVersion);
  console.log(`[DB Migration] 已备份数据库到: ${backupPath}`);
  
  try {
    // 按顺序执行每个版本的迁移
    for (let v = currentVersion; v < targetVersion; v++) {
      const migration = migrations[v];
      console.log(`[DB Migration] 执行 v${v} → v${v + 1}: ${migration.description}`);
      
      await migration.up(db);
      await setVersion(db, v + 1);
      
      console.log(`[DB Migration] v${v + 1} 迁移完成 ✅`);
    }
    
    console.log(`[DB Migration] 全部迁移完成！`);
    return { success: true, isLatest: false, migratedFrom: currentVersion, migratedTo: targetVersion };
    
  } catch (error) {
    console.error(`[DB Migration] 迁移失败:`, error.message);
    
    // 迁移失败，从备份恢复
    console.log(`[DB Migration] 从备份恢复数据库...`);
    await restoreFromBackup(dbPath, backupPath);
    
    return { 
      success: false, 
      error: error.message,
      restored: true
    };
  }
}

module.exports = {
  runMigrations,
  getCurrentVersion,
};
```

### 3.2 在 `main.js` 中集成

```javascript
const { runMigrations } = require('./database-migrator');

function initDatabase() {
  const dbPath = getDatabasePath();
  const db = new sqlite3.Database(dbPath, (err) => {
    if (err) {
      console.error('打开数据库时出错:', err.message);
      return;
    }
    
    // 先创建基础表（保证 v1 表存在）
    createBaseTables(db, async () => {
      // 执行数据库迁移
      const migrationResult = await runMigrations(db, dbPath);
      
      if (!migrationResult.success) {
        // 迁移失败，但已从备份恢复，应用仍可正常使用旧功能
        console.warn('数据库升级失败，但已恢复旧版本');
        // 可选择发送 IPC 通知渲染进程显示提示
      } else if (!migrationResult.isLatest) {
        // 迁移成功，可以通知用户"已升级到新版本"
        console.log(`数据库已从 v${migrationResult.migratedFrom} 升级到 v${migrationResult.migratedTo}`);
      }
      
      console.log('数据库初始化完成');
    });
  });
  
  return db;
}

// 创建 v1 版本的基础表（向后兼容）
function createBaseTables(db, callback) {
  db.run(`CREATE TABLE IF NOT EXISTS study_sessions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    start_time TEXT NOT NULL,
    end_time TEXT,
    duration INTEGER,
    created_at TEXT DEFAULT CURRENT_TIMESTAMP
  )`);
  
  db.run(`CREATE TABLE IF NOT EXISTS user_config (
    key TEXT PRIMARY KEY,
    value TEXT
  )`, callback);
}
```

## 四、各版本迁移明细

### v1 → v2: 标签系统

**新增表**：
- `study_tags` - 标签定义表
- `session_tags` - session 与标签关联表

**新增索引**：
- `idx_study_sessions_start_time` - 优化时间范围查询

**默认数据**：
- 预置 3 个常用标签："通用学习"、"编程"、"阅读"

### v2 → v3: 目标系统

**新增 user_config 键**：
- `daily_goal_seconds` - 每日目标时长
- `weekly_goal_seconds` - 每周目标时长  
- `last_study_date` - 最后学习日期
- `current_streak` - 当前连续天数
- `max_streak` - 历史最长连续天数
- `total_study_seconds` - 累计学习时长

**数据回填**：
```sql
-- 计算历史累计学习时长
UPDATE user_config SET value = (
  SELECT IFNULL(SUM(duration), 0) FROM study_sessions
) WHERE key = 'total_study_seconds';
```

### v3 → v4: 徽章系统

**新增 user_config 键**：
- `earned_badges` - JSON 数组存储已获得的徽章
- `user_level` - 用户等级
- `backup_cards` - 补签卡数量

**数据回填**：
- 根据历史数据自动计算已获得的徽章
- 计算用户当前等级

## 五、平滑升级用户体验

### 5.1 首次启动新版本流程

```
用户升级应用并启动
    ↓
[ 后台 ] 自动执行数据库迁移
    ↓
┌─ 如果迁移成功
│   └─ 正常进入应用，显示"已更新到新版本"提示（可选）
│
└─ 如果迁移失败（极少见）
    ├─ 自动从备份恢复
    ├─ 弹出友好提示："数据库更新遇到问题，已恢复您的学习数据"
    ├─ 所有原有功能完全可用（只是新功能可能不可用）
    └─ 引导用户提交 bug 报告（可选）
```

### 5.2 降级处理（安装旧版本）

如果用户不小心安装了旧版本：

```
旧版本应用启动
    ↓
发现数据库版本 (v3) > 旧版本支持的最高版本 (v2)
    ↓
┌─ 方案 A：拒绝启动，提示用户
│   提示："您安装的版本过旧，请安装最新版本"
│   提供下载链接
│
└─ 方案 B：兼容启动（推荐）
    旧版本只使用自己能理解的表和字段
    新增的表会被忽略，不影响核心功能
    提示："检测到较新的数据库，新功能可能不可用"
```

**推荐方案 B**，因为 SQLite 会自动忽略不存在的列查询，只需在代码中做好空值处理。

### 5.3 多版本兼容性代码示例

```javascript
// renderer.js - 获取连续天数（v3+ 才支持）
async function getStreakDays() {
  try {
    const streak = await window.studyRecord.getConfig('current_streak');
    return streak ? parseInt(streak) : 0;
  } catch (e) {
    // 数据库版本过低，不支持该字段
    return 0;
  }
}

// 使用时
const streakDays = await getStreakDays();
if (streakDays > 0) {
  // 显示连续天数 UI
  showStreakBadge(streakDays);
}
// 否则不显示该功能区域
```

## 六、测试验证方案

### 6.1 迁移测试清单

每次添加新迁移前，必须验证以下场景：

✅ **全新安装测试**
- 删除数据库，启动应用
- 验证所有表正确创建
- 验证功能正常

✅ **跨版本升级测试**
- 准备 v1 数据库（有真实学习数据）
- 升级到最新版本
- 验证所有数据完整
- 验证旧数据正确迁移到新结构

✅ **逐级升级测试**
- v1 → v2 → v3 → v4，每步都验证

✅ **降级兼容测试**
- 用新版本创建 v4 数据库
- 用旧版本代码启动
- 验证核心功能（开始/结束/查看历史）正常

✅ **失败回滚测试**
- 故意在迁移脚本中插入错误
- 验证迁移失败后数据库能从备份恢复
- 验证应用能正常启动并使用原有数据

### 6.2 测试数据准备

在 `test/fixtures/` 目录下保存各版本的测试数据库：
```
test/fixtures/
├── v1_sample.db      # 含 100 条学习记录的 v1 数据库
├── v2_sample.db      # 含标签数据的 v2 数据库
└── v3_sample.db      # 含目标和打卡数据的 v3 数据库
```

## 七、生产环境安全措施

### 7.1 保留备份

升级成功后，备份文件不自动删除，保留 30 天：
```
study_time.db.backup_v1
study_time.db.backup_v2
study_time.db.backup_v3
```

用户可以手动删除，或在应用启动时清理超过 30 天的旧备份。

### 7.2 日志记录

所有迁移操作写入应用日志：
- 迁移开始时间、源版本、目标版本
- 每个迁移步骤的耗时
- 成功/失败状态和错误信息
- 备份文件路径

### 7.3 分批迁移

对于大数据量（>10000 条记录），分批次迁移避免 UI 卡顿。

## 八、总结

这套迁移方案的核心优势：

| 优势 | 说明 |
|------|------|
| **零感知** | 用户完全不知道后台发生了迁移 |
| **绝对安全** | 升级前必备份，失败必回滚，永不丢数据 |
| **兼容降级** | 即使安装旧版本，核心功能依然可用 |
| **易扩展** | 新增版本只需加一个迁移脚本文件 |
| **可测试** | 每个迁移都可独立测试验证 |

**最佳实践**：每次数据库结构变更，必须先写迁移脚本，再写业务代码！
